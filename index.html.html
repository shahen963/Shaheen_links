CREATE TABLE IF NOT EXISTS public.lp_pages (
  id      uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title   text,
  logo    text,
  sub     text,
  bg1     text DEFAULT '#3b0764',
  bg2     text DEFAULT '#1d0535',
  bg3     text DEFAULT '#0a0010',
  links   jsonb DEFAULT '[]'::jsonb,
  updated_at timestamptz DEFAULT now()
);
ALTER TABLE public.lp_pages ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public_read"  ON public.lp_pages FOR SELECT USING (true);
CREATE POLICY "public_write" ON public.lp_pages FOR ALL    USING (true) WITH CHECK (true);
